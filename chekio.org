#找整数最小公倍数
def checkio(data):
    lenth = len(data)
    i = 1
    while True:
        bol = False
        for j in data:
            bol += bool(i%j==0)
        if bol == lenth:
            print(i)
            break
        i+=1
                
        
        
data = (1,2,3,4,5,6,7,8,9)
checkio(data)
